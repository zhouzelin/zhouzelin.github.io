vue-directive-zery

vue自定义指令扩展

快速上手 →

# 安装方便

npm i -S vue-directive-zery
// or
yarn add vue-directive-zery

# 使用简单

import Vue from 'Vue'
import zery from 'vue-directive-zery'
Vue.use(zery)

# 按需加载，防止冲突

import Vue from 'Vue'
import zery from 'vue-directive-zery'
zery.use(['clickOutside', 'watermarker', 'copy', 'debounce', 'limitText'])
Vue.use(zery)
